    subroutine destroy_kkl_grad()
      !< nullify all the gradient pointer setup for k-kL model

      implicit none

      DebugCall('destroy_kkl_grad')

      nullify(gradtk_x)
      nullify(gradtkl_x)

      nullify(gradtk_y)
      nullify(gradtkl_y)

      nullify(gradtk_z)
      nullify(gradtkl_z)

    end subroutine destroy_kkl_grad
