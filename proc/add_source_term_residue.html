    subroutine add_source_term_residue()
      !< Call to add different source terms to the residual of different equations.

      implicit none

      call dmsg(1, 'source', 'add_source_term_residue')

      select case (trim(turbulence))

        case ('none')
          !do nothing
          continue

        case ('sa')
          select case(trim(transition))
            case('none')
              call add_sa_source()
            case('bc')
              call add_saBC_source()
            case DEFAULT
              Fatal_error
          end select

        case ('sst', 'sst2003')
          select case(trim(transition))
            case('none')
              call add_sst_source()
            case('lctm2015')
              call add_sst_source_lctm2015()
            case('bc')
              call add_sst_bc_source()
            case DEFAULT
              Fatal_error
          end select

        case ('kkl')
          call add_kkl_source()

        case DEFAULT
          Fatal_error

      end select

    end subroutine add_source_term_residue
