        subroutine destroy_grid()
            !< Deallocate the memory allocated for the grid.

            implicit none

            call dmsg(1, 'grid', 'destroy_memory')

            call dealloc(grid_x)
            call dealloc(grid_y)
            call dealloc(grid_z)
