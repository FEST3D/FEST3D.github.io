        subroutine destroy_geometry()
          !< Nullify all the face normal pionter 

            implicit none
            
            call dmsg(1, 'geometry', 'destroy_geometry')

            nullify(xnx)
            nullify(xny)
            nullify(xnz)
            nullify(ynx)
            nullify(yny)
            nullify(ynz)
            nullify(znx)
            nullify(zny)
            nullify(znz)
            call deallocate_memory()

        end subroutine destroy_geometry
