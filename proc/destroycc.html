    subroutine destroyCC()
      !< Deallocate memory from the cell-center variables.
      implicit none

      DebugCall("Destroy CC")

      call dealloc(CCnormalX)
      call dealloc(CCnormalY)
      call dealloc(CCnormalZ)
      call dealloc(CCVn)
      call dealloc(DCCVnX)
      call dealloc(DCCVnY)
      call dealloc(DCCVnZ)

    end subroutine destroyCC
